```js
import React from 'react';
import useFileSystem from '@/sukinos/hooks/useFileSystem';
import FileSystemView from './FileSystemView'; // 刚创建的组件
import { FileType } from "@/sukinos/utils/config";

const FileManagerApp = () => {
// 1. 挂载你的钩子逻辑
const {
  state: { items, breadcrumbs, isReady },
  navigation: { loadDir, handleBack, handleRefresh },
  operation: { handleOpenFile } // 按需引入你写好的其他操作
} = useFileSystem('virtual'); // 'virtual' 或者 'local'

// 2. 处理双击事件（视图组件的回调）
const handleItemDoubleClick = (item) => {
    if (item.type === FileType.DIRECTORY) {
      // 如果是文件夹，进入下一级
      loadDir(item.id);
    } else {
      // 如果是文件，执行你的打开逻辑（比如唤起 Notebook 编辑器）
      console.log('打开文件:', item.name);
      // openEditor(item);
    }
};

  // 3. 处理面包屑点击
const handleBreadcrumbClick = (id) => {
    loadDir(id);
};

return (
<div style={{ width: '800px', height: '600px', padding: '20px' }}>
    {/_ 4. 将数据通过 props 喂给视图组件 _/}
    <FileSystemView
      items={items}
      breadcrumbs={breadcrumbs}
      isLoading={!isReady}
      onBack={handleBack}
      onRefresh={handleRefresh}
      onItemDoubleClick={handleItemDoubleClick}
      onItemClick={(item) => console.log('选中了:', item.name)}
      onBreadcrumbClick={handleBreadcrumbClick}
      defaultLayout="list"
      />
  </div>
  );
};

export default FileManagerApp;

```
