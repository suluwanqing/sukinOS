# 这是一个将check和chekGroup封装到card方便使用的组件

## props

| 方法      |  参数   | 说明        |
|options | Arrray    | 用于各项配置 |
|values |当前状态对象|    格式如 { [key]: boolean | any[] } Object {}  |
|onUpdate|状态更新的回调函数 | (key,val)=>void|

## options
key 状态映射的唯一键名 string
type 类型：switch (默认) 或 group string
label 标题 string
desc 辅助描述文字 string
icon 选项图标 (如 MUI Icon) ReactNode
theme 主题颜色 (primary, success, danger 等) string
mode group 专属：选择模式 multiple (默认) 或 single string
subOptions group 专属：子选项配置数组 Option[]
min / max group 专属：选择数量的最小和最大限制 number
```javascript
import CheckCardBar from '@/component/checkCardBar/layout';

const [settings, setSettings] = useState({
  is_vfs: true,
  sync_type: ['cloud'],
  permission_level: ['admin']
});

const handleUpdate = (key, val) => {
setSettings(prev => ({ ...prev, [key]: val }));
};

const SCHEMA = [
// 模式 A：独立开关
{
key: 'is_vfs',
label: '虚拟盘模式',
desc: '开启后将应用挂载至内存虚拟盘中',
icon: <StorageIcon />,
theme: 'success'
},

// 模式 B：互斥单选组
{
key: 'permission_level',
type: 'group',
label: '系统权限级别',
mode: 'single',
min: 1, // 必须选一个，禁止取消
subOptions: [
{ value: 'guest', label: '访客权限', desc: '仅可查看基础资源' },
{ value: 'admin', label: '管理员权限', desc: '拥有完整读写权限' }
]
},

// 模式 C：多选限制组
{
key: 'sync_type',
type: 'group',
label: '数据同步策略',
mode: 'multiple',
max: 2, // 最多选两项
subOptions: [
{ value: 'local', label: '本地备份' },
{ value: 'cloud', label: '云端同步' },
{ value: 'remote', label: '异地容灾', disabled: true }
]
}
];

<CheckCardBar
  options={SCHEMA}
  values={settings}
  onUpdate={handleUpdate}
/>

```
